13def i(arg)
  case arg
    when older.reject(&:my_hair) && years.from_now >= MANY
      send(:valentine, :birthday_greetings, :bottle_of_wine)
  end
  door.lock if Time.now >= Time.parse('2:45')
end

def need(age)
  age <= TOO_OLD
end

alias :feed, :need
